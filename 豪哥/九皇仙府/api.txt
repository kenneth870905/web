权限
	登陆
	/api/auth/login
	方法	POST
	数据	{"name": "", "password": ""}
	管理员的role是ADMIN,普通用户是USER

	更改密码
	/api/auth/password
	方法	POST
	数据	{"password": ""}

用户
	列表
	/api/user
	方法	GET
	参数	page, size

	添加
	/api/user
	方法	POST
	数据	{"name": "", "password": "", "roles": "ADMIN", "status": 0}

	删除
	/api/user/{id}
	方法	DELETE

	更新
	/api/user/{id}
	方法	PUT

园区
	列表
	/api/park
	方法	GET
	参数	page, size

	添加
	/api/park
	方法	POST
	数据	{"name": "", "code": "", "description": ""}

	删除
	/api/park/{id}
	方法	DELETE

	更新
	/api/park/{id}
	方法	PUT

区域单元
	列表
	/api/unit
	方法	GET
	参数	park_id, page, size

	添加
	/api/unit
	方法	POST
	数据	{"name": "", "price": 1, "row_count": 10, "column_count": 10, "status": 0, "description": ""}
	添加区域会根据行数和列数自动添加墓地

	删除
	/api/unit/{id}
	方法	DELETE

	更新
	/api/unit/{id}
	方法	PUT

墓地
	列表
	/api/grave
	方法	GET
	参数	unit_id
	status: 0表示待售 1表示已预订 2表示已交定金 3表示售出

	删除
	/api/grave/{id}
	方法	DELETE

	更新
	/api/grave/{id}
	方法	PUT

订单
	下单
	/api/order
	方法	POST
	数据	{"grave_id": 1}

	订单列表
	/api/order
	方法	GET
	status 1表示下单 2表示已交定金 3表示订单完成售出

	更新订单
	/api/order/{id}
	方法	PUT
	只能更新status

	删除订单
	/api/order/{id}
	方法	DELETE
